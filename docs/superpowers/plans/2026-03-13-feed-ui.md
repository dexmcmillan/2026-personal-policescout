# Feed UI Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the static daily digest with a searchable, chronological card feed combining press releases (last 7 days, from `data/archive/*.json`) and TPS calls for service (last 7 days, from `data/tps_calls.ndjson`), hosted on GitHub Pages.

**Architecture:** A new `build_feed()` function in `scan.py` reads all archive JSON files and the TPS NDJSON, normalizes and filters to last 7 days, merges into one newest-first list, writes `docs/data.json`, and renders `docs/index.html` from a new `templates/feed.html` Jinja2 template. The HTML shell loads `docs/data.json` via `fetch()` and renders cards + live search with inline vanilla JS and CSS. `main()` calls `build_feed()` at the very end, after all existing scraping logic (unchanged). `templates/digest.html` is deleted.

**Tech Stack:** Python 3.12, `jinja2` (already a dependency), vanilla JS/CSS, GitHub Pages static hosting

```python
def normalize_date(date_str: str | None) -> str | None:
    """
    Convert a date string in any known format to ISO YYYY-MM-DD.
    Returns None if the input is None or cannot be parsed.
    Handles: "2026-03-13", "March 13, 2026", "Mar 13, 2026", "13 March 2026", etc.
    """
    if not date_str:
        return None
    # ISO format (also handles datetimes — take first 10 chars)
    try:
        return datetime.fromisoformat(date_str[:10]).date().isoformat()
    except ValueError:
        pass
    # Human-readable formats
    for fmt in ("%b %d, %Y", "%B %d, %Y", "%d %B %Y", "%B %d %Y", "%b %d %Y"):
        try:
            return datetime.strptime(date_str.strip()[:20], fmt).date().isoformat()
        except ValueError:
            continue
    return None
```

```python
# --- Feed builder ---


def _load_archive_items(archive_dir: Path, cutoff: date) -> list[dict]:
    """
    Load all press release items from archive/*.json files.
    Normalizes dates to ISO YYYY-MM-DD, filters to on/after cutoff.
    Malformed JSON files are skipped with a warning.
    """
    items = []
    for path in archive_dir.glob("*.json"):
        try:
            raw = json.loads(path.read_text(encoding="utf-8"))
        except Exception as e:
            print(f"  [build_feed] WARNING: skipping malformed archive {path.name}: {e}")
            continue
        for entry in raw:
            iso_date = normalize_date(entry.get("date"))
            if iso_date is None or iso_date >= cutoff.isoformat():
                title = (entry.get("title") or "").lower()
                source = (entry.get("service_name") or "").lower()
                # search_text = space-joined lowercase title + source
                items.append({
                    "type": "press_release",
                    "title": entry.get("title", ""),
                    "url": entry.get("url"),
                    "date": iso_date,
                    "source": entry.get("service_name", ""),
                    "search_text": " ".join(filter(None, [title, source])),
                    "_sort_key": iso_date or "",
                })
    return items


def _load_tps_items(tps_ndjson: Path, cutoff: date) -> list[dict]:
    """
    Load TPS call records from the NDJSON log.
    Returns empty list with a warning if the file does not exist.
    """
    if not tps_ndjson.exists():
        print(f"  [build_feed] WARNING: TPS NDJSON not found at {tps_ndjson}, skipping TPS data")
        return []
    items = []
    with tps_ndjson.open(encoding="utf-8") as f:
        for lineno, line in enumerate(f, 1):
            line = line.strip()
            if not line:
                continue
            try:
                rec = json.loads(line)
            except Exception as e:
                print(f"  [build_feed] WARNING: skipping malformed TPS line {lineno}: {e}")
                continue
            occurred_at = rec.get("occurred_at")
            if not occurred_at:
                continue  # skip TPS records with no timestamp
            iso_date = occurred_at[:10]  # "2026-03-13T15:18:56+00:00" → "2026-03-13"
            if iso_date < cutoff.isoformat():
                continue
            call_type = (rec.get("call_type") or "").lower()
            division = (rec.get("division") or "").lower()
            cross_streets = (rec.get("cross_streets") or "").lower()
            # search_text = space-joined lowercase: call_type + "toronto police service" + division + cross_streets
            items.append({
                "type": "tps_call",
                "title": rec.get("call_type", ""),
                "url": None,
                "date": iso_date,
                "occurred_at": occurred_at,
                "source": "Toronto Police Service",
                "division": rec.get("division", ""),
                "cross_streets": rec.get("cross_streets", ""),
                "search_text": " ".join(filter(None, [
                    call_type,
                    "toronto police service",
                    division,
                    cross_streets,
                ])),
                "_sort_key": occurred_at,  # full ISO timestamp; sorts precisely within a day
            })
            # The output dict includes both "occurred_at" (spec-required field) AND "_sort_key"
            # (internal, stripped before writing). They have the same value but different keys.
    return items


def build_feed(
    archive_dir: Path,
    tps_ndjson: Path,
    output_dir: Path,
    days: int = 7,
) -> None:
    """
    Build the card feed: merge press releases + TPS calls, write docs/data.json
    and render docs/index.html from templates/feed.html.

    Errors (missing files, malformed JSON) are logged and skipped gracefully.
    """
    cutoff = date.today() - timedelta(days=days)
    print(f"  [build_feed] Cutoff: {cutoff} ({days} days)")

    press_items = _load_archive_items(archive_dir, cutoff)
    print(f"  [build_feed] Press releases in window: {len(press_items)}")

    tps_items = _load_tps_items(tps_ndjson, cutoff)
    print(f"  [build_feed] TPS calls in window: {len(tps_items)}")

    all_items = press_items + tps_items
    all_items.sort(key=lambda x: x["_sort_key"], reverse=True)

    # Strip internal sort key before writing
    for item in all_items:
        item.pop("_sort_key", None)

    output_dir.mkdir(parents=True, exist_ok=True)
    data_path = output_dir / "data.json"
    data_path.write_text(
        json.dumps(all_items, ensure_ascii=False, indent=None),
        encoding="utf-8",
    )
    print(f"  [build_feed] Wrote {data_path} ({len(all_items)} items)")

    # Render index.html
    generated_at = datetime.now(timezone.utc).strftime("%B %d, %Y at %H:%M UTC")
    env = Environment(loader=FileSystemLoader(str(TEMPLATE_DIR)), autoescape=True)
    try:
        template = env.get_template("feed.html")
    except Exception as e:
        print(f"  [build_feed] WARNING: could not load feed.html template: {e}")
        return
    html = template.render(generated_at=generated_at)
    index_path = output_dir / "index.html"
    index_path.write_text(html, encoding="utf-8")
    print(f"  [build_feed] Wrote {index_path}")
```

```html
    (function () {
      var allItems = [];

      // --- Time formatting ---
      function formatTime(isoStr) {
        // isoStr: "2026-03-13T15:18:56+00:00"
        var d = new Date(isoStr);
        var now = new Date();
        // Compare calendar dates in UTC
        var dDate = d.toISOString().slice(0, 10);
        var todayDate = now.toISOString().slice(0, 10);
        var yd = new Date(now);
        yd.setUTCDate(yd.getUTCDate() - 1);
        var yesterdayDate = yd.toISOString().slice(0, 10);

        var hh = d.getUTCHours();
        var mm = d.getUTCMinutes();
        var ampm = hh >= 12 ? 'PM' : 'AM';
        var h12 = hh % 12 || 12;
        var mStr = mm < 10 ? '0' + mm : String(mm);
        var timeStr = h12 + ':' + mStr + ' ' + ampm;

        if (dDate === todayDate) return timeStr + ' today';
        if (dDate === yesterdayDate) return 'yesterday ' + timeStr;

        var months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
        return months[d.getUTCMonth()] + ' ' + d.getUTCDate() + ' at ' + timeStr;
      }

      // Format date-only string "2026-03-13" → "Mar 13, 2026"
      function formatDate(isoDate) {
        if (!isoDate) return '';
        var parts = isoDate.split('-');
        var months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
        return months[parseInt(parts[1], 10) - 1] + ' ' + parseInt(parts[2], 10) + ', ' + parts[0];
      }

      // --- Card rendering ---
      function renderCard(item) {
        var div = document.createElement('div');

        if (item.type === 'press_release') {
          div.className = 'card card-pr';
          div.innerHTML =
            '<div class="card-meta">' +
              '<span class="card-source">' + esc(item.source) + '</span>' +
              '<span class="card-time">' + esc(formatDate(item.date)) + '</span>' +
            '</div>' +
            '<div class="card-title">' +
              (item.url
                ? '<a href="' + esc(item.url) + '" target="_blank" rel="noopener">' + esc(item.title) + ' →</a>'
                : esc(item.title)) +
            '</div>';
        } else {
          // tps_call
          div.className = 'card card-tps card-type-tps';
          var divText = item.division ? 'TPS · ' + item.division : 'TPS';
          var timeText = item.occurred_at ? formatTime(item.occurred_at) : formatDate(item.date);
          div.innerHTML =
            '<div class="card-meta">' +
              '<span class="card-source">' + esc(divText) + '</span>' +
              '<span class="card-time">' + esc(timeText) + '</span>' +
            '</div>' +
            '<div class="card-title">' + esc(item.title) + '</div>' +
            (item.cross_streets
              ? '<div class="card-location">' + esc(item.cross_streets) + '</div>'
              : '');
        }

        return div;
      }

      function esc(str) {
        if (!str) return '';
        return String(str)
          .replace(/&/g, '&amp;')
          .replace(/</g, '&lt;')
          .replace(/>/g, '&gt;')
          .replace(/"/g, '&quot;');
      }

      // --- Render visible cards ---
      function renderFeed(query) {
        var feed = document.getElementById('feed');
        feed.innerHTML = '';
        var q = query ? query.toLowerCase().trim() : '';
        var visible = 0;

        for (var i = 0; i < allItems.length; i++) {
          var item = allItems[i];
          var show = !q || item.search_text.indexOf(q) !== -1;
          if (show) {
            feed.appendChild(renderCard(item));
            visible++;
          }
        }

        var countEl = document.getElementById('count');
        if (q) {
          countEl.textContent = 'Showing ' + visible + ' of ' + allItems.length;
        } else {
          countEl.textContent = allItems.length + ' items';
        }

        if (visible === 0) {
          var empty = document.createElement('p');
          empty.style.color = '#666';
          empty.style.padding = '20px 0';
          empty.textContent = 'No results for "' + query + '"';
          feed.appendChild(empty);
        }
      }

      // --- Live search ---
      document.getElementById('search').addEventListener('input', function () {
        renderFeed(this.value);
      });

      // --- Load data ---
      fetch('data.json')
        .then(function (r) {
          if (!r.ok) throw new Error('HTTP ' + r.status);
          return r.json();
        })
        .then(function (data) {
          allItems = data;
          renderFeed('');
        })
        .catch(function (err) {
          document.getElementById('feed').innerHTML =
            '<div id="error">Failed to load data: ' + esc(String(err)) + '</div>';
        });
    })();
```
